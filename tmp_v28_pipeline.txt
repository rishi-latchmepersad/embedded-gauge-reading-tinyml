"""Helpers for replaying localizer-driven crops into the exact V28 reader."""

from __future__ import annotations

from dataclasses import dataclass
import math

import numpy as np
from numpy.typing import NDArray

from embedded_gauge_reading_tinyml.board_crop_compare import resize_with_pad_rgb
from embedded_gauge_reading_tinyml.geometry_cascade import (
    crop_box_from_center,
    source_xy_from_resized_xy,
)
from embedded_gauge_reading_tinyml.firmware_preprocessing import (
    firmware_training_crop_box,
)


@dataclass(frozen=True, slots=True)
class LocalizerCropDecision:
    """One crop-box decision returned by a localizer head."""

    crop_box_xyxy: tuple[float, float, float, float]
    crop_source: str
    accepted: bool
    fallback_reason: str | None
    details: dict[str, float]


def _clamp_norm(value: float) -> float:
    """Clamp a normalized coordinate into the [0, 1] interval."""
    return max(0.0, min(1.0, value))


def _keypoint_center_from_coords(
    keypoint_coords: NDArray[np.float32],
    *,
    center_keypoint_index: int | None = None,
) -> tuple[float, float]:
    """Compute the keypoint center from a `(K, 2)` coordinate tensor.

    If center_keypoint_index is provided, returns that specific keypoint.
    Otherwise falls back to the mean of all keypoints (legacy behavior).
    """
    coords = np.asarray(keypoint_coords, dtype=np.float32).reshape(-1, 2)
    if coords.size == 0:
        raise ValueError("keypoint_coords must contain at least one keypoint.")
    if center_keypoint_index is not None:
        if center_keypoint_index >= coords.shape[0]:
            raise ValueError(f"center_keypoint_index {center_keypoint_index} out of range for {coords.shape[0]} keypoints.")
        return (float(coords[center_keypoint_index, 0]), float(coords[center_keypoint_index, 1]))
    return (float(np.mean(coords[:, 0])), float(np.mean(coords[:, 1])))


def decode_keypoint_crop_box(
    keypoint_coords: NDArray[np.float32],
    *,
    source_width: int,
    source_height: int,
    input_size: int = 224,
    keypoint_crop_scale: float = 0.83,
    min_crop_size: float = 48.0,
    keypoint_center_x_bias_pixels: float = 0.0,
    keypoint_center_y_bias_pixels: float = 0.0,
    center_keypoint_index: int | None = None,
) -> LocalizerCropDecision:
    """Convert keypoints into a source-space crop around the predicted center."""
    if keypoint_crop_scale <= 0.0:
        raise ValueError("keypoint_crop_scale must be > 0.")
    if min_crop_size <= 0.0:
        raise ValueError("min_crop_size must be > 0.")

    center_x_canvas, center_y_canvas = _keypoint_center_from_coords(keypoint_coords, center_keypoint_index=center_keypoint_index)
    center_x_source, center_y_source = source_xy_from_resized_xy(
        (center_x_canvas, center_y_canvas),
        crop_box_xyxy=(0.0, 0.0, float(source_width), float(source_height)),
        image_height=source_height,
        image_width=source_width,
    )
    center_x_source += keypoint_center_x_bias_pixels
    center_y_source += keypoint_center_y_bias_pixels

    base_crop_box = firmware_training_crop_box(source_width, source_height)
    crop_box_xyxy = crop_box_from_center(
        center_xy=(center_x_source, center_y_source),
        base_crop_box_xyxy=base_crop_box,
        image_shape=(source_height, source_width),
        shrink_scale=keypoint_crop_scale,
        min_size=min_crop_size,
    )

    crop_width = crop_box_xyxy[2] - crop_box_xyxy[0]
    crop_height = crop_box_xyxy[3] - crop_box_xyxy[1]
    details = {
        "center_x_canvas": center_x_canvas,
        "center_y_canvas": center_y_canvas,
        "center_x_source": center_x_source,
        "center_y_source": center_y_source,
        "crop_width": crop_width,
        "crop_height": crop_height,
        "crop_scale": keypoint_crop_scale,
        "min_crop_size": min_crop_size,
    }
    return LocalizerCropDecision(
        crop_box_xyxy=crop_box_xyxy,
        crop_source="keypoint_coords",
        accepted=True,
        fallback_reason=None,
        details=details,
    )


def decode_source_crop_box(
    source_crop_box: NDArray[np.float32],
    *,
    source_width: int,
    source_height: int,
    input_size: int = 224,
) -> LocalizerCropDecision:
    """Convert a direct source-space xyxy prediction into a crop decision."""
    if np.asarray(source_crop_box).size < 4:
        raise ValueError("source_crop_box must contain four values.")

    crop = np.asarray(source_crop_box, dtype=np.float32).reshape(-1)
    x0_norm = _clamp_norm(float(crop[0]))
    y0_norm = _clamp_norm(float(crop[1]))
    x1_norm = _clamp_norm(float(crop[2]))
    y1_norm = _clamp_norm(float(crop[3]))

    x0 = x0_norm * float(source_width)
    y0 = y0_norm * float(source_height)
    x1 = x1_norm * float(source_width)
    y1 = y1_norm * float(source_height)

    if x1 <= x0:
        x1 = min(float(source_width), x0 + 1.0)
    if y1 <= y0:
        y1 = min(float(source_height), y0 + 1.0)

    crop_box_xyxy = (
        max(0.0, min(x0, float(source_width))),
        max(0.0, min(y0, float(source_height))),
        max(1.0, min(x1, float(source_width))),
        max(1.0, min(y1, float(source_height))),
    )
    crop_width = crop_box_xyxy[2] - crop_box_xyxy[0]
    crop_height = crop_box_xyxy[3] - crop_box_xyxy[1]
    details = {
        "x0_norm": x0_norm,
        "y0_norm": y0_norm,
        "x1_norm": x1_norm,
        "y1_norm": y1_norm,
        "crop_width": crop_width,
        "crop_height": crop_height,
        "input_size": float(input_size),
    }
    return LocalizerCropDecision(
        crop_box_xyxy=crop_box_xyxy,
        crop_source="source_crop_box",
        accepted=True,
        fallback_reason=None,
        details=details,
    )


def decode_heatmap_crop_box(
    keypoint_heatmaps: NDArray[np.float32],
    *,
    source_width: int,
    source_height: int,
    input_size: int = 224,
    keypoint_crop_scale: float = 0.83,
    min_crop_size: float = 48.0,
    keypoint_center_x_bias_pixels: float = 0.0,
    keypoint_center_y_bias_pixels: float = 0.0,
) -> LocalizerCropDecision:
    """Convert keypoint heatmaps into a crop around the predicted center."""
    from embedded_gauge_reading_tinyml.geometry_cascade import heatmaps_to_coords

    coords = heatmaps_to_coords(np.asarray(keypoint_heatmaps, dtype=np.float32))
    return decode_keypoint_crop_box(
        coords,
        source_width=source_width,
        source_height=source_height,
        input_size=input_size,
        keypoint_crop_scale=keypoint_crop_scale,
        min_crop_size=min_crop_size,
        keypoint_center_x_bias_pixels=keypoint_center_x_bias_pixels,
        keypoint_center_y_bias_pixels=keypoint_center_y_bias_pixels,
    )


def resize_full_frame(
    source_image: NDArray[np.uint8],
    *,
    image_size: int = 224,
) -> NDArray[np.uint8]:
    """Resize the full source frame onto the square model canvas."""
    return resize_with_pad_rgb(
        source_image,
        (
            0.0,
            0.0,
            float(source_image.shape[1]),
            float(source_image.shape[0]),
        ),
        image_size=image_size,
    )


def clamp_angle_radians(angle_rad: float) -> float:
    """Normalize an angle into the `[0, 2π)` interval."""
    normalized = math.fmod(angle_rad, 2.0 * math.pi)
    if normalized < 0.0:
        normalized += 2.0 * math.pi
    return normalized
